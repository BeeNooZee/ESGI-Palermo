Algorithme TP1_2
Début
    Afficher "TP1_2"
    
    Déclarer les variables j, entier, R1, R2 comme entiers
    R1 <- 0
    R2 <- 1
    
    Afficher "Entrer un entier :"
    Lire entier
    
    Pour j allant de 1 à entier faire
        R1 <- R1 + j
        Afficher j
        Si j = entier alors
            Afficher "="
        Sinon
            Afficher "+"
        Fin Si
    Fin Pour
    
    Afficher R1
    
    Pour j allant de 1 à entier faire
        R2 <- R2 * j
        Afficher j
        Si j = entier alors
            Afficher "="
        Sinon
            Afficher "*"
        Fin Si
    Fin Pour
    
    Afficher R2
Fin
